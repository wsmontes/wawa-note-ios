✅ Proposta: AgentOrchestrator com modos

Novo: Um orquestrador, dois modos, tools por contexto
AgentOrchestrator (actor)
├─ aiService: AIService
├─ toolRegistry: AgentToolRegistry
├─ contextManager: ContextWindowManager
│
├─ runInteractive(session:history:context:) → AsyncStream<AgentEvent>
│   Modo chat. Tools: ShellTool + contexto do projeto/item ativo.
│
├─ runAutonomous(task:item:project:) → AgentResult
│   Modo pipeline. Tools: ShellTool + WriteAnalysis + SelectSchema +
│                        SelectSkill + SetTitle + WriteSpeakers.
│
└─ tools(for: AgentMode) → [AgentTool]
    Resolve tools baseado no modo + schema do projeto.

enum AgentMode: String, Codable {
    case chat
    case analysis          // pipeline de ingestão
    case projectIngestion   // ingestão em projeto
    case graphDiscovery     // descoberta de conexões
}

Implementação
actor AgentOrchestrator {
    let aiService: AIService
    let toolRegistry: AgentToolRegistry
    let contextManager: ContextWindowManager
    let configProvider: AIConfigProvider

    func runInteractive(
        session: ChatSession,
        history: [AIMessage],
        context: ToolContext
    ) -> AsyncThrowingStream<AgentEvent, Error> {
        let mode = AgentMode.chat
        let tools = toolRegistry.tools(for: mode)
        let systemPrompt = configProvider.systemPrompt(for: mode.rawValue)
        let messages = contextManager.buildMessages(system: systemPrompt, history: history)

        return aiService.sendStreaming(
            feature: mode.featureKey,
            messages: messages,
            tools: tools,
            toolChoice: "auto"
        )
        // Tool loop é interno ao AgentOrchestrator
        // Caller só vê a stream de eventos (texto, tool_call, tool_result)
    }

    func runAutonomous(
        task: AgentTask,
        item: KnowledgeItem,
        project: Project?
    ) async throws -> AgentResult {
        let mode = task.mode  // .analysis, .projectIngestion, .graphDiscovery
        let tools = toolRegistry.tools(for: mode)
        // ... loop autônomo com max iterations ...
    }
}
