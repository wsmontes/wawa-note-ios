🔌 Protocolos internos (trocáveis, testáveis)

3 protocolos + 1 type
protocol AIConfigProvider: Sendable {
    func requestParams(for feature: String, model: String) -> AIFeatureParams
    func modelFor(feature: String) -> String
    func resolvedModelFor(feature: String, context: ModelContext) -> String?
    func presetFor(model: String) -> ModelPreset?
}

protocol ProviderResolver: Sendable {
    func resolve(for feature: String, context: ProviderContext) throws -> any AIProvider
    func resolveStreaming(for feature: String, context: ProviderContext) throws -> any AIProvider
    var activeProviderID: String { get }
}

protocol ModelPolicy: Sendable {
    func selectModel(for feature: String, budget: BudgetTracker, context: ModelContext) -> String
    func availableModels(for provider: any AIProvider) async -> [String]
}

struct AIServiceConfig: Sendable {
    var retryPolicy: RetryPolicy
    var circuitBreaker: CircuitBreaker?
    var defaultFeature: String  // fallback feature
}

🏗️ AIService — implementação esqueleto

Como o AIService monta a request
actor AIService {
    let configProvider: AIConfigProvider
    let providerResolver: ProviderResolver
    let modelPolicy: ModelPolicy
    let retryConfig: RetryPolicy

    func send(feature: String, messages: [AIMessage], ...) async throws -> AIResponse {
        // 1. Model selection via policy
        let model = modelPolicy.selectModel(for: feature, budget: budget)

        // 2. Params from config (temperature, maxTokens) — SEMPRE
        let params = configProvider.requestParams(for: feature, model: model)

        // 3. Provider resolved with health + fallback
        let provider = try providerResolver.resolve(for: feature, context: ctx)

        // 4. Build internal AIRequest (callers never see this)
        let internalRequest = AIRequest(
            model: model, messages: messages,
            temperature: params.temperature,
            maxTokens: params.maxTokens, ...
        )

        // 5. Send with retry + circuit breaker
        return try await retryPolicy.execute {
            try await provider.send(internalRequest)
        }
    }
}
