# Project Creation & Update — User Journeys Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Unified project creation/update API (`ProjectService.create` + `update`), redesigned creation UX (2 entry points + AI-assisted promote), inline editing on Project Home, and proactive agent suggestions.

**Architecture:** Centralize all project mutations through `ProjectService.create()` and `ProjectService.update()` with field-level provenance and automatic snapshotting. Migrate ~12 scattered callers progressively. Add `CreateProjectSheet` (replaces current minimal sheet), evolve `PromoteToProjectSheet` with AI assistance, add collapsible "Project Info" section to `ProjectHomeView` with inline editors, and introduce `ProjectSuggestion` model for agent-driven proactive suggestions.

**Tech Stack:** Swift 6, SwiftData (`@Model`), SwiftUI, `@MainActor` services, `async/await`, `FieldAuthorityService` (existing), `VersioningService` (existing).

## Global Constraints

- Target: iPhone 14 Plus (iOS 18.6.2), also tested on simulator (iOS 26.5)
- `FieldOrigin` enum uses `.llm` (not `.agent`) — match existing code
- `ProjectService` is `@MainActor` — all methods are MainActor-isolated
- `Project` is a SwiftData `@Model` class — mutations persist via `context.save()`
- New SwiftData models must be registered in `ModelContainer` (WawaNoteApp.swift)
- New files must be added to `project.pbxproj` (via Xcode)
- Follow existing patterns: thin views, service-layer logic, dependency injection via initializers
- Existing `ProjectService.create(name:summary:iconName:)` already exists — extend, don't replace
- `VersioningService.createSnapshot(projectID:label:trigger:context:)` already exists
- `FieldAuthorityService` + `FieldProvidence` protocol already exist — reuse

```swift
import Foundation

/// Batch of optional field changes for a project update.
/// Only non-nil fields are applied. Nil fields are skipped.
struct ProjectUpdateFields: Sendable {
    var name: String?
    var summary: String?
    var intention: String?
    var customInstructions: String?
    var colorHex: String?
    var iconName: String?
    var status: ProjectStatus?
    var frameworkId: String?
    var holdIngestionForDoubts: Bool?

    /// Returns true if at least one field is non-nil.
    var hasChanges: Bool {
        name != nil || summary != nil || intention != nil ||
        customInstructions != nil || colorHex != nil || iconName != nil ||
        status != nil || frameworkId != nil || holdIngestionForDoubts != nil
    }
}
```

```swift
/// Apply a batch of field changes to a project with field-level validation,
/// provenance tracking, and automatic snapshotting.
/// - Parameters:
///   - id: Project UUID
///   - fields: Batch of optional changes — only non-nil fields are applied
///   - origin: Who is making the change (.user, .llm, .system)
///   - reason: Optional human-readable reason for audit trail
/// - Returns: The updated project
func update(id: UUID, fields: ProjectUpdateFields, origin: FieldOrigin, reason: String? = nil) throws -> Project {
    guard let project = try fetch(id: id) else {
        throw ProjectServiceError.projectNotFound(id)
    }
    guard fields.hasChanges else { return project }

    var provenance = project.provenance
    var changed = false

    // Name
    if let name = fields.name, FieldAuthorityService.shared.canModify(field: "name", of: project, by: origin) {
        project.name = name
        project.slug = name.lowercased().replacingOccurrences(of: " ", with: "-")
        project.nameIsAutoGenerated = false
        provenance.mark(field: "name", origin: origin)
        changed = true
    }

    // Summary
    if let summary = fields.summary, FieldAuthorityService.shared.canModify(field: "summary", of: project, by: origin) {
        project.summary = summary
        provenance.mark(field: "summary", origin: origin)
        changed = true
    }

    // Intention
    if let intention = fields.intention, FieldAuthorityService.shared.canModify(field: "intention", of: project, by: origin) {
        project.intention = intention
        project.intentionIsAutoGenerated = false
        provenance.mark(field: "intention", origin: origin)
        changed = true
    }

    // Custom instructions
    if let ci = fields.customInstructions, FieldAuthorityService.shared.canModify(field: "customInstructions", of: project, by: origin) {
        project.customInstructions = ci
        provenance.mark(field: "customInstructions", origin: origin)
        changed = true
    }

    // Color
    if let hex = fields.colorHex, FieldAuthorityService.shared.canModify(field: "colorHex", of: project, by: origin) {
        project.colorHex = hex
        provenance.mark(field: "colorHex", origin: origin)
        changed = true
    }

    // Icon
    if let icon = fields.iconName, FieldAuthorityService.shared.canModify(field: "iconName", of: project, by: origin) {
        project.iconName = icon
        provenance.mark(field: "iconName", origin: origin)
        changed = true
    }

    // Status
    if let status = fields.status, FieldAuthorityService.shared.canModify(field: "status", of: project, by: origin) {
        project.status = status
        provenance.mark(field: "status", origin: origin)
        changed = true
    }

    // Framework
    if let fid = fields.frameworkId, FieldAuthorityService.shared.canModify(field: "frameworkId", of: project, by: origin) {
        project.frameworkId = fid
        provenance.mark(field: "frameworkId", origin: origin)
        changed = true
    }

    // Hold ingestion
    if let hold = fields.holdIngestionForDoubts, FieldAuthorityService.shared.canModify(field: "holdIngestionForDoubts", of: project, by: origin) {
        project.holdIngestionForDoubts = hold
        provenance.mark(field: "holdIngestionForDoubts", origin: origin)
        changed = true
    }

    guard changed else { return project }

    project.updatedAt = Date()
    project.fieldProvenanceJSON = provenance.encode()

    try context.save()

    // Snapshot after save succeeds
    VersioningService.shared.createSnapshot(
        projectID: id,
        label: reason ?? "update via \(origin.rawValue)",
        trigger: .manual,
        context: context
    )

    return project
}
```

```swift
func create(
    name: String,
    template: ProjectTemplate? = nil,
    sourceItemIDs: [UUID] = [],
    origin: FieldOrigin = .user
) throws -> Project {
    let project = Project(name: name, colorHex: assignColor())
    project.nameIsAutoGenerated = false

    var provenance = project.provenance
    provenance.mark(field: "name", origin: origin)
    project.fieldProvenanceJSON = provenance.encode()

    // Apply template
    if let template {
        project.frameworkId = template.frameworkId
        provenance.mark(field: "frameworkId", origin: origin)
        project.fieldProvenanceJSON = provenance.encode()
    }

    context.insert(project)
    try context.save()

    // Link source items
    for itemID in sourceItemIDs {
        try? addItem(itemID, to: project.id)
    }

    // Snapshot
    VersioningService.shared.createSnapshot(
        projectID: project.id,
        label: "created: \(name)",
        trigger: .manual,
        context: context
    )

    return project
}
```

```swift
struct CreateProjectSheet: View {

                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("Done") { createProject() }
                        .disabled(name.trimmingCharacters(in: .whitespaces).isEmpty)
                }
            }
            .sheet(isPresented: $showItemPicker) {
                // Multi-select of unassigned items — deferred to Task 7
                Text("Item picker — coming in Promote flow")
            }
        }
    }

    private func createProject() {
        let trimmed = name.trimmingCharacters(in: .whitespaces)
        guard !trimmed.isEmpty else { return }

        _ = try? service.create(
            name: trimmed,
            template: selectedTemplate,
            sourceItemIDs: sourceItemIDs,
            origin: .user

        dismiss()
    }
```

```swift
enum SuggestionType: String, Codable, CaseIterable, Sendable {
    case summaryUpdate
    case taskCreate
    case riskAlert
    case connectionProposal

    case pending
    case accepted
    case dismissed

@Model
final class ProjectSuggestion {
    var id: UUID
    var projectID: UUID
    var title: String
    var body: String
    var suggestionTypeRaw: String
    var proposedFieldsJSON: String?
    var statusRaw: String
    var createdAt: Date
    var resolvedAt: Date?

    var suggestionType: SuggestionType {
        get { SuggestionType(rawValue: suggestionTypeRaw) ?? .summaryUpdate }
        set { suggestionTypeRaw = newValue.rawValue }
    }

    var status: SuggestionStatus {
        get { SuggestionStatus(rawValue: statusRaw) ?? .pending }
        set { statusRaw = newValue.rawValue }
    }

    init(
        projectID: UUID,
        title: String,
        body: String,
        suggestionType: SuggestionType,
        proposedFields: ProjectUpdateFields? = nil,
        status: SuggestionStatus = .pending,
```

```swift
import Foundation
import SwiftData

@MainActor
final class ProjectSuggestionService {
    private let context: ModelContext

    init(context: ModelContext) {
        self.context = context
    }

    /// Create a new pending suggestion. Deduplicates — if a similar pending
    /// suggestion exists for the same project+type, skips creation.
    func emit(
        projectID: UUID,
        title: String,
        body: String,
        type: SuggestionType,
        proposedFields: ProjectUpdateFields? = nil
    ) {
        // Dedup: check for existing pending suggestion of same type for this project
        let existing = try? context.fetch(
            FetchDescriptor<ProjectSuggestion>(
                predicate: #Predicate {
                    $0.projectID == projectID &&
                    $0.suggestionTypeRaw == type.rawValue &&
                    $0.statusRaw == SuggestionStatus.pending.rawValue
                }
            )
        )
        if let existing, !existing.isEmpty { return }

        let suggestion = ProjectSuggestion(
            projectID: projectID,
            title: title,
            body: body,
            suggestionType: type,
            proposedFields: proposedFields
        )
        context.insert(suggestion)
        try? context.save()
    }

    /// Accept a suggestion — apply its proposed fields and mark as accepted.
    func accept(_ suggestion: ProjectSuggestion) throws {
        if let json = suggestion.proposedFieldsJSON,
           let data = json.data(using: .utf8),
           let fields = try? JSONDecoder().decode(ProjectUpdateFields.self, from: data) {
            _ = try ProjectService(context: context).update(
                id: suggestion.projectID,
                fields: fields,
                origin: .llm,
                reason: "accepted suggestion: \(suggestion.title)"
            )
        }
        suggestion.status = .accepted
        suggestion.resolvedAt = Date()
        try context.save()
    }

    /// Dismiss a suggestion without applying its changes.
    func dismiss(_ suggestion: ProjectSuggestion) throws {
        suggestion.status = .dismissed
        suggestion.resolvedAt = Date()
        try context.save()
    }

    /// Fetch pending suggestions for a project, ordered by recency.
    func pending(for projectID: UUID, limit: Int = 2) -> [ProjectSuggestion] {
        let descriptor = FetchDescriptor<ProjectSuggestion>(
            predicate: #Predicate {
                $0.projectID == projectID &&
                $0.statusRaw == SuggestionStatus.pending.rawValue
            },
            sortBy: [SortDescriptor(\.createdAt, order: .reverse)]
        )
        let results = (try? context.fetch(descriptor)) ?? []
        return Array(results.prefix(limit))
    }

    /// Expire suggestions older than 7 days that are still pending.
    func expireOldSuggestions() {
        let cutoff = Date().addingTimeInterval(-7 * 24 * 3600)
        let descriptor = FetchDescriptor<ProjectSuggestion>(
            predicate: #Predicate {
                $0.statusRaw == SuggestionStatus.pending.rawValue &&
                $0.createdAt < cutoff
            }
        )
        guard let expired = try? context.fetch(descriptor) else { return }
        for suggestion in expired {
            suggestion.status = .dismissed
            suggestion.resolvedAt = Date()
        }
        try? context.save()
    }
}
```
